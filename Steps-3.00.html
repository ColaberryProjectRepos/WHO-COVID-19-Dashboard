Calculated Columns 3

Steps

  1.  Right-click on countries-aggregated on the right-hand side underneath Fields and click "New column"
  2.  Type into the bar the measures below that you will use for Date variables

 

Calculated Columns (Column)

lastWeekTotalCases = Var __CountryName= 'countries-aggregated'[Country]
                     Var __LastWeek= DATEADD('countries-aggregated'[Date],-7,DAY)
                     Return
            CALCULATE(
             MAX('countries-aggregated'[Confirmed]) ,
            FILTER(
             'countries-aggregated',
             'countries-aggregated'[Date] = __LastWeek &&
             'countries-aggregated'[Country] = __CountryName
                    )
                )+0

 

lastWeekTotalDeaths = Var __CountryName= 'countries-aggregated'[Country]
                     Var __LastWeek= DATEADD('countries-aggregated'[Date],-7,DAY)
                     Return
            CALCULATE(
             MAX('countries-aggregated'[Deaths]) ,
            FILTER(
             'countries-aggregated',
             'countries-aggregated'[Date] = __LastWeek &&
             'countries-aggregated'[Country] = __CountryName
                    )
                )+0

 

lastWeekTotalRecovered = Var __CountryName= 'countries-aggregated'[Country]
                     Var __LastWeek= DATEADD('countries-aggregated'[Date],-7,DAY)
                     Return
            CALCULATE(
             SUM('countries-aggregated'[Recovered]) ,
            FILTER(
             'countries-aggregated',
             'countries-aggregated'[Date] = __LastWeek &&
             'countries-aggregated'[Country] = __CountryName
                    )
                )+0
